135 static Mesh shadeMesh(Mesh mesh)
136 {
137  vector<Triangle> triangles;
138  triangles.reserve(mesh->size());
139  VectorF lightVector(1, 1, 1);
140  lightVector = normalize(lightVector);
141  VectorF specularityLightVector = normalize(lightVector + VectorF(0, 0, 1));
142  for(Triangle tri : *mesh)
143  {
144  VectorF normal = tri.normal();
145  float v = dot(normal, lightVector);
146  if(v < 0)
147  v = 0;
148  Color c = Color::HSB(0, 1, min(1.0, 0.3 * v + 0.15 + 0.5 * pow(max<float>(0.0, dot(normal, specularityLightVector)), 16)));
149  tri.c[0] = scale(c, tri.c[0]);
150  tri.c[1] = scale(c, tri.c[1]);
151  tri.c[2] = scale(c, tri.c[2]);
152  triangles.push_back(tri);
153  }
154  return Mesh(new Mesh_t(mesh->texture(), triangles));
155 }
