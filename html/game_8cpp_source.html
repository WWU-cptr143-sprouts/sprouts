106  * this if statement is only true if on a border, since conn 1

133  if (isInArea(*areas[j],Coord(original.x, original.y+1)))
134  tempSets[1]->push_back(areas[j]);

192  * Look at horizontal lines and see if drawing a vertical line from the current
193  * point to that line would cross it. If we cross an odd number of lines within
194  * an area in the one direction (here, up), we're in the area since we only

197 bool Game::isInArea(const Area& target, Coord position) const //Blame Luke for any problems here
198 {
199  int tSize = target.size(), lCount = 0, lSize = 0;
200 
201  for(int i=0;i<tSize;i++)
202  {
203  lSize = target[i]->line->size();
204 
205  // We are starting from 1 since a line is two between two coordinates.
206  // We'll use the index and index-1 for those two.
207  for(int j=1;j<lSize;j++)
208  {
209  /*
210  * If this was change to y instead of x then it would find
211  * horizontal lines
212  */
213  const Line& lRef = *(target[i]->line);
214 
215  // A Vertical line to the left of this point.
216  if (lRef[j-1].x == lRef[j].x && lRef[j].x < position.x)
217  {
218  int minY,maxY;
219  minY=min(lRef[j].y,lRef[j-1].y);
220  maxY=max(lRef[j].y,lRef[j-1].y);
221 
222  // Only one of these can have a equal sign. If a node is on the
223  // start/end of a line. We only want to detect one of these since
224  // otherwise it'll cross the other (odd/even) lines the wrong
225  // number of times.
226  if (position.y <= maxY && position.y > minY)
227  ++lCount;
228  }
229  }
230  }
231 
232  // If it is in the area then lCount, the number of lines between it and the
233  // origin, will be odd, and thus return true.
234  return lCount%2;
235 }

347 void Game::doMove(const Line& line, Coord middle, bool extraChecks)
348 {
349  updated = false;
350 
351  // Determine the end nodes
352  Node* a = NULL;
353  Node* b = NULL;
354 
355  for (int i = 0; i < nodes.size(); i++)
356  {
357  if (nodes[i]->getLoci() == line.front())
358  {
359  if (a)
360  throw InvalidNode();
361  else
362  a = nodes[i];
363  }
364 
365  if (nodes[i]->getLoci() == line.back())
366  {
367  if (b)
368  throw InvalidNode();
369  else
370  b = nodes[i];
371  }
372  }
373 
374  // Couldn't find nodes
375  if (!a || !b)
376  throw InvalidNode();
377 
378  if (line.size() == 0)
379  throw InvalidLine(line);
380 
381  // Now that line crossings work, we don't really need to do this. In games
382  // with 10 nodes or so, it starts taking a while. We'll specify when
383  // creating the game whether we want extra checks (like this) to be
384  // performed.
385  if (extraChecks)
386  {
387  updateAreas();
388 
389  if (!connectable(*a, *b))
390  throw NotConnectable();
391  }
392 
393  // Split the line using the middle coordinate
394  int count = 0; // Add to first line when 0, second when 1
395  Line AC_line;
396  Line CB_line;
397 
398  // First point since we start at the second point, looking at the first
399  // line segment
400  AC_line.push_back(line.front());
401 
402  // When between two points, split there
403  for (int i = 1; i < line.size(); i++)
404  {
405  if ((middle.y == line[i].y && line[i].y == line[i-1].y && // Horizontal
406  ((middle.x > line[i].x && middle.x < line[i-1].x) ||
407  (middle.x < line[i].x && middle.x > line[i-1].x))) ||
408  (middle.x == line[i].x && line[i].x == line[i-1].x && // Vertical
409  ((middle.y > line[i].y && middle.y < line[i-1].y) ||
410  (middle.y < line[i].y && middle.y > line[i-1].y))))
411  {
412  ++count;
413  AC_line.push_back(middle);
414  CB_line.push_back(middle);
415  CB_line.push_back(line[i]);
416  }
417  else
418  {
419  if (count == 0)
420  AC_line.push_back(line[i]);
421  else
422  CB_line.push_back(line[i]);
423  }
424  }
425 
426  // We should have found a place to put the middle point
427  if (count != 1)
428  throw InvalidMiddle(count, middle);
429 
430  // Note that if the middle point is on a corner, it will throw above because
431  // we check that x or y is less than one and greater than the other, meaning
432  // that it can't be equal to either.
433 
434  Line& AC = insertLine(AC_line);
435  Line& CB = insertLine(CB_line);
436 
437  try
438  {
439  // Insert middle node C between nodes A and B
440  Node& c = insertNode(middle,
441  Connection(&AC, a),
442  Connection(&CB, b));
443  try
444  {
445 
446  // Likewise add the connection to the start and end nodes
447  a->addConnection(Connection(&AC, &c));
448  b->addConnection(Connection(&CB, &c));
449 
450  ++moveCount;
451  }
452  catch (...)
453  {
454  // Get rid of the node we just added
455  deleteLastNode();
456  throw;
457  }
458 
459  }
460  // Since it ran into problems, make sure we get rid of anything we added
461  catch (...)
462  {
463  // We definitely inserted lines, so get rid of them.
464  delete &AC;
465  delete &CB;
466  lines.pop_back();
467  lines.pop_back();
468  throw;
469  }
470 
471  updateAreas();
472 }
