115 void Node::updateOpen()
116 {
117  int count = 0;
118 
119  // Blank it again
120  fill(&open[0], &open[0]+4, true); // By default every direction is open
121 
122  // For each of the connections, set open[dir] to false
123  for (int i = 0; i < 3; i++)
124  {
125  if (connections[i].exists())
126  {
127  const Coord* other;
128 
129  // Keep track so we can check if it's valid afterwards
130  ++count;
131 
132  // Note that this can be simplified if we always make sure a line
133  // ends with the node pointed to by dest, but until then, check
134  // based on coordinates
135  const Line& line = *(connections[i].line);
136 
137  // A line must be at least the beginning and ending node
138  if (line.size() < 2)
139  throw InvalidLine(line);
140 
141  if (line.front() == loci) // At beginning
142  other = &line[1];
143  else if (line.back() == loci) // At end
144  other = &line[line.size()-2];
145  else // In the middle? It should be at the beginning or end!
146  throw InvalidLine(line);
147 
148  // It can't be the same point
149  if (loci == *other)
150  throw InvalidLine(line);
151 
152  // Determine direction
153  if (loci.x == other->x) // Vertical
154  {
155  if (loci.y < other->y) // Down
156  {
157  if (open[Down])
158  open[Down] = false;
159  else
160  throw NodeEntryCollision();
161  }
162  else // Up
163  {
164  if (open[Up])
165  open[Up] = false;
166  else
167  throw NodeEntryCollision();
168  }
169  }
170  else if (loci.y == other->y) // Horizontal
171  {
172  if (loci.x < other->x) // Right
173  {
174  if (open[Right])
175  open[Right] = false;
176  else
177  throw NodeEntryCollision();
178  }
179  else // Left
180  {
181  if (open[Left])
182  open[Left] = false;
183  else
184  throw NodeEntryCollision();
185  }
186  }
187  else // Neither, so invalid
188  {
189  throw InvalidLine(line);
190  }
191  }
192  }
193 
194  // If there's only two, they must be 180 degrees from each other
195  if (count == 2 &&
196  !((open[Left] == false && open[Right] == false) ||
197  (open[Up] == false && open[Down] == false)))
198  throw InvalidCorner();
199 }
