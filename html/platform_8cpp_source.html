756 static Event *makeEvent()
757 {
758  static bool needCharEvent = false;
759  static wstring characters;
760  if(needCharEvent)
761  {
762  needCharEvent = false;
763  wchar_t character = characters[0];
764  characters.erase(0, 1);
765  return new KeyPressEvent(character);
766  }
767  while(true)
768  {
769  SDL_Event SDLEvent;
770  if(SDL_PollEvent(&SDLEvent) == 0)
771  {
772  return nullptr;
773  }
774  switch(SDLEvent.type)
775  {
776  case SDL_KEYDOWN:
777  {
778  KeyboardKey key = translateKey(SDLEvent.key.keysym.scancode);
779  Event *retval = new KeyDownEvent(key, translateModifiers((SDL_Keymod)SDLEvent.key.keysym.mod), keyState(key));
780  keyState(key) = true;
781  return retval;
782  }
783  case SDL_KEYUP:
784  {
785  KeyboardKey key = translateKey(SDLEvent.key.keysym.scancode);
786  Event *retval = new KeyUpEvent(key, translateModifiers((SDL_Keymod)SDLEvent.key.keysym.mod));
787  keyState(key) = false;
788  return retval;
789  }
790  case SDL_MOUSEMOTION:
791  return new MouseMoveEvent(SDLEvent.motion.x, SDLEvent.motion.y, SDLEvent.motion.xrel, SDLEvent.motion.yrel);
792  case SDL_MOUSEBUTTONDOWN:
793  {
794  MouseButton button = translateButton(SDLEvent.button.button);
795  buttonState = static_cast<MouseButton>(buttonState | button); // set bit
796  return new MouseDownEvent(SDLEvent.button.x, SDLEvent.button.y, 0, 0, button);
797  }
798  case SDL_MOUSEBUTTONUP:
799  {
800  MouseButton button = translateButton(SDLEvent.button.button);
801  buttonState = static_cast<MouseButton>(buttonState & ~button); // clear bit
802  return new MouseUpEvent(SDLEvent.button.x, SDLEvent.button.y, 0, 0, button);
803  }
804  case SDL_JOYAXISMOTION:
805  case SDL_JOYBALLMOTION:
806  case SDL_JOYHATMOTION:
807  case SDL_JOYBUTTONDOWN:
808  case SDL_JOYBUTTONUP:
809  //TODO (jacob#): handle joysticks
810  break;
811  case SDL_QUIT:
812  return new QuitEvent();
813  case SDL_SYSWMEVENT:
